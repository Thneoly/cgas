#### 3.1.2 检测算法

```python
# 单 IP 高频访问检测算法
def detect_high_frequency_access(ip, window_size=60, threshold=1000):
    """
    检测单 IP 高频访问
    
    Args:
        ip: 源 IP 地址
        window_size: 时间窗口 (秒)
        threshold: 阈值 (请求数/窗口)
    
    Returns:
        bool: 是否检测到高频访问
    """
    # 获取时间窗口内请求数
    request_count = get_request_count(ip, window_size)
    
    # 计算请求频率
    frequency = request_count / window_size * 60  # 转换为每分钟
    
    # 判断是否超过阈值
    if frequency > threshold:
        log_alert(f"高频访问检测：IP={ip}, 频率={frequency}/min")
        return True
    
    return False
```

#### 4.1.2 检测算法

```python
# 权限提升尝试检测算法
def detect_privilege_escalation(user_id, resource, action):
    """
    检测权限提升尝试
    
    Args:
        user_id: 用户 ID
        resource: 访问资源
        action: 操作类型
    
    Returns:
        bool: 是否检测到权限提升尝试
    """
    # 获取用户当前权限
    user_permissions = get_user_permissions(user_id)
    
    # 检查是否有所需权限
    required_permission = f"{resource}:{action}"
    if required_permission not in user_permissions:
        # 记录尝试
        log_attempt(user_id, resource, action)
        
        # 检查尝试频率
        attempt_count = get_attempt_count(user_id, window=300)  # 5 分钟
        if attempt_count > 3:
            log_alert(f"权限提升尝试：user={user_id}, attempts={attempt_count}")
            return True
    
    return False
```
